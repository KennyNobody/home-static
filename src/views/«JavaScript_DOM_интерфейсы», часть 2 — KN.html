Найти min\max


var inputData = '1, -5.8 или 10, хотя 34 + -5.3 и 73, тоже может 37 или -123 я хз точно';

function getMinMax() {
    // Превращаем строку в массив
    let minNumber = 0;
    let maxNumber = 0;
	let dataArray = inputData.split(' ');

    //Задействуем метод массива .filter
	let dataNumber =  dataArray.filter(function(number) {
	   // Преобразовывываем все значения к числу
	   let newNumber = parseInt(number, 10);
	   
	   if (typeof newNumber == "number") {
	       return newNumber;
	   };
	   
    });
    
    // Устанавливаем значение min и max
    dataNumber.forEach(function(item, i, dataNumber) {
        if (item > maxNumber) {
            maxNumber = +item;
        }
        
        if (item < minNumber) {
            minNumber = +item;
        } 
    });
    
    // Выводим результат
    console.log("min: " + minNumber + ', max: ' + maxNumber);
}

getMinMax();

// "min: -123, max: 37"
